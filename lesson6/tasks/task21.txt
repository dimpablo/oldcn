<script>
    const pairs = [
        { char: "不", meaning: "отрицание, вызывает инверсию дополнения" },
        { char: "我", meaning: "местоимение, при 不 — инвертируется перед глаголом" },
        { char: "有", meaning: "иметь (в значении 'наличие')" },
        { char: "亡", meaning: "не иметь (отрицание вместо 不有)" },
        { char: "及", meaning: "предлог «к моменту...», «когда наступит...»" }
    ];
    let selectedWord = null;
    let matchedPairs = 0;
    const wordList = document.getElementById("words");
    const defList = document.getElementById("defs");
    const resultEl = document.getElementById("result");

    function shuffle(array) {
        const arr = [...array];
        for (let i = arr.length - 1; i > 0; i--) {
            const j = Math.floor(Math.random() * (i + 1));
            [arr[i], arr[j]] = [arr[j], arr[i]];
        }
        return arr;
    }

    function init() {
        wordList.innerHTML = '';
        defList.innerHTML = '';
        selectedWord = null;
        matchedPairs = 0;
        resultEl.textContent = '';

        const shuffledChars = shuffle(pairs);
        const shuffledMeanings = shuffle(pairs.map(p => p.meaning));

        shuffledChars.forEach(item => {
            const el = document.createElement("div");
            el.className = "word-item";
            el.dataset.char = item.char;
            el.textContent = item.char;
            el.onclick = () => selectWord(el);
            wordList.appendChild(el);
        });

        shuffledMeanings.forEach(meaning => {
            const el = document.createElement("div");
            el.className = "def-item";
            el.dataset.meaning = meaning;
            el.textContent = meaning;
            el.onclick = () => selectDefinition(el);
            defList.appendChild(el);
        });
    }

    function selectWord(el) {
        if (el.classList.contains("matched") || selectedWord) return;
        if (selectedWord) selectedWord.classList.remove("selected");
        selectedWord = el;
        el.classList.add("selected");
    }

    function selectDefinition(el) {
        if (el.classList.contains("matched")) return;
        if (!selectedWord) {
            resultEl.textContent = "Сначала выберите иероглиф.";
            setTimeout(() => resultEl.textContent = "", 1500);
            return;
        }
        const char = selectedWord.dataset.char;
        const meaning = el.dataset.meaning;
        const correctMeaning = pairs.find(p => p.char === char).meaning;
        if (meaning === correctMeaning) {
            selectedWord.classList.add("matched");
            el.classList.add("matched");
            selectedWord.onclick = null;
            el.onclick = null;
            selectedWord = null;
            matchedPairs++;
            if (matchedPairs === 5) {
                resultEl.innerHTML = "✅ Все верно! Вы усвоили ключевые элементы Урока 6.";
                resultEl.className = "success";
                sendResult(true);
            }
        } else {
            resultEl.textContent = "❌ Ошибка.";
            setTimeout(() => {
                resultEl.textContent = "";
            }, 1500);
            selectedWord.classList.remove("selected");
            selectedWord = null;
        }
    }

    function reset() {
        init();
    }

    function sendResult(success) {
        window.parent.postMessage({
            type: "TASK_RESULT",
            success: !!success,
            message: success ? "Все соответствия верны" : "Есть ошибки"
        }, "*");
    }

    init();
    window.parent.postMessage({ type: "TASK_EXISTS", valid: true }, "*");
</script>
