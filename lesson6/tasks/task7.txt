<h2>Почему в фразе «帝不我降熯» местоимение «我» стоит перед глаголом?</h2>
<p><em>帝不我降熯 — «Не пошлёт ли Небесный Владыка нам засуху?»</em></p>
<div id="answers"></div>
<button id="check-btn">Проверить</button>
<p id="result" class="message"></p>
<script>
    const options = [
        { text: "При отрицании 不 и личном дополнении (我, 余) — инверсия обязательна", correct: true },
        { text: "Местоимение всегда стоит перед глаголом", correct: false },
        { text: "Это вопрос с 其, поэтому порядок изменён", correct: false },
        { text: "Глагол 降 требует инверсии", correct: false }
    ];

    function render() {
        const container = document.getElementById("answers");
        [...options].sort(() => Math.random() - 0.5).forEach(opt => {
            const label = document.createElement("label");
            const input = document.createElement("input");
            input.type = "radio";
            input.name = "q";
            input.dataset.correct = opt.correct;
            label.append(input, document.createTextNode(opt.text));
            container.appendChild(label);
        });
    }

    document.getElementById("check-btn").addEventListener("click", () => {
        const selected = document.querySelector('input[name="q"]:checked');
        const result = document.getElementById("result");
        if (!selected) {
            result.textContent = "Выберите вариант.";
            return;
        }
        if (selected.dataset.correct === "true") {
            result.innerHTML = "✅ Верно! При 不 + местоимение — инверсия дополнения обязательна.";
            sendResult(true);
        } else {
            result.textContent = "❌ Нет. Это грамматическое правило, а не выбор автора.";
            sendResult(false);
        }
    });

    function sendResult(success) {
        window.parent.postMessage({ type: "TASK_RESULT", success }, "*");
    }

    render();
    window.parent.postMessage({ type: "TASK_EXISTS", valid: true }, "*");
</script>
